from os import walk

# Parse image directories
datadir = "data/cubicasa5k/"
categories = ["colorful", "high_quality", "high_quality_architectural"]
image_dirs = {cat: None for cat in categories} # list of dir names in each category
for cat in categories:
    path = datadir + cat
    for _, dirnames, _ in walk(path):
        image_dirs[cat] = dirnames
        break

def plot_hist(ax, data,xlabel, bins, range_=None, title=None):
    """Helper function for plotting histogram 
    """
    ax.hist(data, bins=bins, range=range_)
    ax.set_xlabel(xlabel)
    ax.set_ylabel("counts")
    ax.set_title(title)

# Get image sizes
original_sizes = {cat: [] for cat in categories}
scaled_sizes = {cat: [] for cat in categories}

for cat, dirs in image_dirs.items():
    # Read original and scaled image sizes
    for dir_ in dirs:
        orig_fpath = datadir + cat + "/" + dir_ + "/F1_original.png"
        orig_shape = imagesize.get(orig_fpath)
        original_sizes[cat].append((int(dir_), *orig_shape))
        
        scaled_fpath = datadir + cat + "/" + dir_ + "/F1_scaled.png"
        scaled_shape = imagesize.get(scaled_fpath)
        scaled_sizes[cat].append((int(dir_), *scaled_shape))

    # Store as numpy array
    original_sizes[cat] = np.array(original_sizes[cat])
    scaled_sizes[cat] = np.array(scaled_sizes[cat])

# Plot aspect ratios histograms
for cat in categories:
    # Get aspect ratios
    original_aspects = original_sizes[cat][:, 2] / original_sizes[cat][:, 1]
    scaled_aspects = scaled_sizes[cat][:, 2] / scaled_sizes[cat][:, 1]
    assert np.allclose(original_aspects, scaled_aspects, atol=3e-3)
    
    # Plot histogram
    fig, ax = plt.subplots(nrows=1, ncols=1, figsize=(15, 5))
    plot_hist(ax, original_aspects, xlabel="aspect ratio", bins=40, range_=(0, 3.5))

print("Aspect ratios for each original image and the corresponding scaled image are the same.")

Aspect ratios for each original image and the corresponding scaled image are the same.

The aspect ratios of the images vary a lot within a category. However, the distributions of the aspect ratios are fairly similar among the 3 categories.

for cat in categories:
    # Get aspect ratios
    original_pixels = original_sizes[cat][:, 2] * original_sizes[cat][:, 1] / 1e6
    scaled_pixels = scaled_sizes[cat][:, 2] * scaled_sizes[cat][:, 1] / 1e6
    assert np.allclose(original_aspects, scaled_aspects, atol=3e-3)
    
    # Plot histogram
    fig, axes = plt.subplots(nrows=1, ncols=2, figsize=(15, 5))
    fig.suptitle(f"# of Pixels for {cat}")
    plot_hist(axes[0], original_pixels, xlabel="# of pixels (million)", bins=40, title="original")
    plot_hist(axes[1], scaled_pixels, xlabel="# of pixels (million)", bins=40, title="scaled")

Compared to aspect ratios, the image sizes (i.e. numbers of pixels) are more similar across images. Scaled images, however, have wider distributions of image sizes than original images.
